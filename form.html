const express = require('express'); const mysql = require('mysql2'); const app = express(); app.use(express.json()); const db = mysql.createConnection({ host: 'mysql-service', user: 'webuser', password: 'webpass', database: 'webdb' }); app.post('/save', (req, res) => { const { name, country, email, mobile, nationality, marital_status } = req.body; const sql = `INSERT INTO users (name, country, email, mobile, nationality, marital_status) VALUES (?, ?, ?, ?, ?, ?)`; db.query(sql, [name, country, email, mobile, nationality, marital_status], (err, result) => { if (err) { console.error(err); res.status(500).send("Database error"); } else { res.send("Data saved successfully!"); } }); }); app.listen(3000, () => { console.log('Server running on port 3000'); });
